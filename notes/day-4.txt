SOLUTION — APIRouter
Definition: Splits your API into separate files — one file per feature.

student_api/

routers/students.py
from fastapi import APIRouter
from fastapi.responses import JSONResponse
from pydantic import BaseModel, Field
from typing import Optional

router = APIRouter(
    prefix="/students",       # all routes start with /students
    tags=["Students"]         # groups in /docs
)

# Fake DB
db = []
counter = 1

class StudentCreate(BaseModel):
    name: str  = Field(min_length=2)
    age: int   = Field(ge=1, le=100)
    city: str
    course: str

class StudentUpdate(BaseModel):
    name:   Optional[str] = None
    age:    Optional[int] = None
    city:   Optional[str] = None
    course: Optional[str] = None

@router.get("/")
def get_all():
    return {"students": db, "total": len(db)}

@router.post("/", status_code=201)
def create(student: StudentCreate):
    global counter
    new = student.dict()
    new["id"] = counter
    db.append(new)
    counter += 1
    return {"message": "Created", "data": new}

@router.get("/{student_id}")
def get_one(student_id: int):
    for s in db:
        if s["id"] == student_id:
            return s
    return JSONResponse(status_code=404, content={"error": "Not found"})

@router.put("/{student_id}")
def update(student_id: int, student: StudentCreate):
    for i, s in enumerate(db):
        if s["id"] == student_id:
            db[i] = {"id": student_id, **student.dict()}
            return {"message": "Updated", "data": db[i]}
    return JSONResponse(status_code=404, content={"error": "Not found"})

@router.patch("/{student_id}")
def partial_update(student_id: int, updates: StudentUpdate):
    for i, s in enumerate(db):
        if s["id"] == student_id:
            db[i].update(updates.dict(exclude_none=True))
            return {"message": "Updated", "data": db[i]}
    return JSONResponse(status_code=404, content={"error": "Not found"})

@router.delete("/{student_id}")
def delete(student_id: int):
    for i, s in enumerate(db):
        if s["id"] == student_id:
            removed = db.pop(i)
            return {"message": "Deleted", "deleted": removed}
    return JSONResponse(status_code=404, content={"error": "Not found"})

HTTPException — BETTER ERROR HANDLING
Definition: Raises HTTP errors properly — cleaner than JSONResponse.
from fastapi import APIRouter, HTTPException

router = APIRouter(prefix="/students", tags=["Students"])

db = [{"id": 1, "name": "Nikhil", "age": 21}]

@router.get("/{student_id}")
def get_one(student_id: int):
    for s in db:
        if s["id"] == student_id:
            return s

    # Clean way to raise error
    raise HTTPException(
        status_code=404,
        detail="Student not found"
    )

# Other common HTTPExceptions:
raise HTTPException(status_code=400, detail="Bad request")
raise HTTPException(status_code=401, detail="Not authenticated")
raise HTTPException(status_code=403, detail="Not authorized")
raise HTTPException(status_code=409, detail="Already exists")
raise HTTPException(status_code=500, detail="Server error")
HTTPException vs JSONResponse:
JSONResponse  → manual, more control
HTTPException → cleaner, recommended by FastAPI
CUSTOM ERROR MESSAGES
from fastapi import HTTPException

def find_student(student_id: int, db: list):
    for s in db:
        if s["id"] == student_id:
            return s
    raise HTTPException(
        status_code=404,
        detail={
            "error": "Student not found",
            "student_id": student_id,
            "message": "Check ID and try again"
        }
    )

@router.get("/{student_id}")
def get_one(student_id: int):
    return find_student(student_id, db)
GLOBAL EXCEPTION HANDLER
from fastapi import FastAPI, Request
from fastapi.responses import JSONResponse

app = FastAPI()

# Catches any unhandled exception
@app.exception_handler(Exception)
async def global_exception_handler(request: Request, exc: Exception):
    return JSONResponse(
        status_code=500,
        content={
            "error": "Internal server error",
            "message": str(exc)
        }
    )

# Catches 404 globally
@app.exception_handler(404)
async def not_found_handler(request: Request, exc):
    return JSONResponse(
        status_code=404,
        content={"error": "Route not found"}
    )
VALIDATION ERROR HANDLER — CUSTOM 422 MESSAGE
from fastapi.exceptions import RequestValidationError

@app.exception_handler(RequestValidationError)
async def validation_handler(request: Request, exc: RequestValidationError):
    return JSONResponse(
        status_code=422,
        content={
            "error": "Validation failed",
            "details": exc.errors()
        }
    )
    
COMMON MISTAKES
Mistake 1 — Forgetting init.py:
routers/students.py  → exists
routers/__init__.py  → missing
→ ImportError when running

Fix: create empty __init__.py inside routers/
Mistake 2 — Wrong import path:
# WRONG — if running from inside routers/
from students import router

# RIGHT — always run from project root
from routers import students
Mistake 3 — raise vs return for errors:
# WRONG — HTTPException must be raised
return HTTPException(status_code=404)   # does nothing

# RIGHT
raise HTTPException(status_code=404, detail="Not found")
